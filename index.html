import java.util.*; public class HelloWorld { public static void main (String[] args) { System.out.println("Try and guess the number!"); Random r=new Random(); int number=r.nextInt(5); //System.out.println(number); Scanner s=new Scanner(System.in); int amazing = 1; int good = 5; int count = 0; while (true){ count = count +1; Long guess = s.nextLong(); if (guess > number){ System.out.println("Too High"); } else if (guess < number) { System.out.println("Too Low"); } else{ System.out.println("You got it!" + count + "tries"); if (count == amazing) { System.out.println("AMAZING"); } else if (amazing < count && count < good +1 ) { System.out.println("Good job"); } else{ System.out.println("You're a horrible guesser"); } break; } } } }
